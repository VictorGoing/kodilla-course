select I.SUMMARY, I2.NAME from ISSUES I
JOIN issueslists I2 on i2.ID = I.ISSUESLIST_ID;

select I.SUMMARY, U.FIRSTNAME, U.LASTNAME from ISSUES I
JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO;

select U.FIRSTNAME, U.LASTNAME, COUNT(*) AS QUESTCOUNT from ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID;
